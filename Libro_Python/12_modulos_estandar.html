os
os · os.path
sistema de archivos · rutas · directorios · variables de entorno

Interacción con el sistema operativo: navegar directorios, crear/eliminar carpetas, leer variables de entorno, y construir rutas de forma compatible entre Windows/Mac/Linux.

os — sistema de archivos y directorios
import os

# directorio actual
os.getcwd()                        # → '/home/lian/proyectos'

# cambiar directorio
os.chdir("/home/lian")

# listar archivos de un directorio
os.listdir(".")                   # → ['main.py', 'datos', 'README.md']
os.listdir("/home/lian")

# crear directorio
os.mkdir("nueva_carpeta")         # falla si ya existe
os.makedirs("a/b/c", exist_ok=True) # crea árbol, no falla si existe

# eliminar
os.remove("archivo.txt")          # elimina archivo
os.rmdir("carpeta_vacia")         # elimina carpeta vacía

# renombrar / mover
os.rename("viejo.txt", "nuevo.txt")

# variables de entorno
os.environ["PATH"]                # leer
os.getenv("HOME", "/default")    # leer con valor por defecto
os.environ["MI_VAR"] = "valor"   # definir

# ejecutar comando del sistema
os.system("ls -la")               # retorna código de salida

# recorrer árbol de directorios
for raiz, dirs, archivos in os.walk("."):
    for arch in archivos:
        print(os.path.join(raiz, arch))
os.path — rutas multiplataforma
import os

# unir rutas (usa / en Linux/Mac, \ en Windows automáticamente)
os.path.join("/home", "lian", "archivo.txt")
# → '/home/lian/archivo.txt'

# verificar existencia
os.path.exists("archivo.txt")    # → True / False
os.path.isfile("archivo.txt")    # es archivo?
os.path.isdir("carpeta")         # es directorio?

# partes de una ruta
ruta = "/home/lian/datos/archivo.txt"
os.path.basename(ruta)            # → 'archivo.txt'
os.path.dirname(ruta)             # → '/home/lian/datos'
os.path.splitext("foto.jpg")      # → ('foto', '.jpg')
os.path.split(ruta)               # → ('/home/lian/datos', 'archivo.txt')

# ruta absoluta desde relativa
os.path.abspath(".")             # → ruta completa del directorio actual

# tamaño en bytes
os.path.getsize("archivo.txt")   # → 1024
Usa siempre os.path.join() para construir rutas — nunca concatenes strings con "/" manualmente. En Windows las rutas usan "\" y tu código se rompe. os.path.join elige el separador correcto automáticamente.
math
math
funciones matemáticas · constantes · trigonometría · logaritmos
Constantes y funciones esenciales
import math

# constantes
math.pi      # → 3.141592653589793
math.e       # → 2.718281828459045
math.inf     # → inf  (infinito)
math.tau     # → 6.28...  (2π)

# redondeo
math.floor(3.9)     # → 3   (hacia abajo)
math.ceil(3.1)      # → 4   (hacia arriba)
math.trunc(3.9)     # → 3   (trunca, igual que int())

# potencias y raíces
math.sqrt(25)        # → 5.0
math.pow(2, 10)      # → 1024.0   (igual que 2**10 pero retorna float)
math.isqrt(17)       # → 4   (raíz entera — piso de sqrt)

# logaritmos
math.log(100, 10)    # → 2.0  (log base 10 de 100)
math.log(math.e)     # → 1.0  (log natural = log base e)
math.log10(1000)    # → 3.0
math.log2(8)         # → 3.0
math.exp(1)          # → 2.718... (e^1)

# valor absoluto y signo
math.fabs(-5.5)      # → 5.5  (como abs pero siempre float)
math.copysign(3, -1)# → -3.0 (toma signo del segundo argumento)

# factores
math.factorial(5)    # → 120  (5! = 5×4×3×2×1)
math.gcd(12, 8)      # → 4   (máximo común divisor)
math.lcm(4, 6)       # → 12  (mínimo común múltiplo) — Python 3.9+

# comparación de floats
math.isclose(0.1+0.2, 0.3)       # → True
math.isnan(float("nan"))          # → True
math.isinf(float("inf"))          # → True

# trigonometría (en radianes)
math.sin(math.pi / 2)   # → 1.0
math.cos(0)            # → 1.0
math.degrees(math.pi) # → 180.0  (rad → grados)
math.radians(180)     # → 3.14... (grados → rad)
date
datetime
fechas · horas · diferencias · formateo · timedelta
Las 4 clases principales del módulo
from datetime import date, time, datetime, timedelta

# date — solo fecha (año, mes, día)
hoy       = date.today()               # fecha de hoy
nacimiento = date(2000, 6, 15)          # fecha específica
print(hoy.year, hoy.month, hoy.day)    # atributos

# time — solo hora (sin fecha)
t = time(14, 30, 0)                    # 14:30:00
print(t.hour, t.minute, t.second)

# datetime — fecha Y hora combinadas
ahora  = datetime.now()              # fecha y hora actuales
evento = datetime(2026, 12, 31, 23, 59, 59)

# timedelta — duración / diferencia entre fechas
diff = date.today() - nacimiento    # resta dos fechas → timedelta
print(diff.days)                      # cuántos días han pasado

una_semana = timedelta(weeks=1)
proximo    = date.today() + una_semana # sumar tiempo a una fecha
ayer       = date.today() - timedelta(days=1)
Formateo — strftime y strptime
from datetime import datetime

ahora = datetime.now()

# strftime — datetime → string con formato
ahora.strftime("%d/%m/%Y")         # → '12/04/2026'
ahora.strftime("%H:%M:%S")         # → '14:30:00'
ahora.strftime("%d de %B de %Y")  # → '12 de April de 2026'
ahora.strftime("%A")               # → 'Sunday'  (día de la semana)

# strptime — string → datetime (parsear)
datetime.strptime("12/04/2026", "%d/%m/%Y")
datetime.strptime("2026-04-12 14:30", "%Y-%m-%d %H:%M")

# isoformat — formato estándar ISO 8601
ahora.isoformat()                   # → '2026-04-12T14:30:00.123456'
datetime.fromisoformat("2026-04-12T14:30:00")  # parsear ISO

# códigos de formato más usados
# %Y año 4 dígitos  %m mes 01-12   %d día 01-31
# %H hora 00-23     %M minuto      %S segundo
# %A día semana     %B mes nombre  %y año 2 dígitos

random
números aleatorios · elecciones · mezclar · semilla
Funciones principales
import random

# float entre 0.0 y 1.0 (exclusivo)
random.random()               # → 0.37444...  (varía cada vez)

# float en un rango
random.uniform(1.5, 5.5)      # → 3.24...  (entre 1.5 y 5.5)

# entero en un rango (ambos extremos incluidos)
random.randint(1, 6)          # → simula un dado
random.randrange(0, 100, 5) # → múltiplo de 5 entre 0 y 99

# elegir de una secuencia
random.choice(["roca", "papel", "tijera"])   # un elemento
random.choices(["a","b","c"], k=3)            # 3 con repetición
random.choices(["cara","cruz"], weights=[70,30], k=5) # con pesos
random.sample([1,2,3,4,5], k=3)            # 3 SIN repetición

# mezclar lista in-place
baraja = list(range(1, 53))
random.shuffle(baraja)        # modifica la lista, no retorna nada

# semilla — reproducibilidad (para tests)
random.seed(42)
random.random()               # siempre el mismo valor con seed 42
random no es criptográficamente seguro. Para tokens, contraseñas o cualquier cosa de seguridad usa secrets en vez de random: import secrets; secrets.token_hex(16)

json
serializar · deserializar · leer/escribir archivos JSON

JSON es el formato estándar para intercambiar datos entre sistemas. Python convierte automáticamente entre tipos Python y tipos JSON.

dumps / loads — convertir entre Python y string JSON
import json

datos = {
    "nombre": "Lian",
    "edad": 22,
    "activo": True,
    "notas": [15, 18, 20]
}

# dumps — dict Python → string JSON
json_str = json.dumps(datos)
# '{"nombre": "Lian", "edad": 22, "activo": true, "notas": [15, 18, 20]}'

json_str = json.dumps(datos, indent=2)   # con sangría (legible)
json_str = json.dumps(datos, ensure_ascii=False) # preserva ñ, tildes

# loads — string JSON → dict Python
recuperado = json.loads(json_str)
print(recuperado["nombre"])   # → Lian
dump / load — leer y escribir archivos JSON
import json

# escribir a archivo
datos = {"usuario": "lian", "nivel": 5}

with open("config.json", "w", encoding="utf-8") as f:
    json.dump(datos, f, indent=2, ensure_ascii=False)

# leer desde archivo
with open("config.json", encoding="utf-8") as f:
    config = json.load(f)

print(config["nivel"])   # → 5

# manejo de errores — JSON malformado
try:
    json.loads("esto no es json")
except json.JSONDecodeError as e:
    print(f"JSON inválido: {e}")
sys
sys
argumentos · salida · versión · rutas
Lo más usado de sys
import sys

# argumentos de línea de comandos
# si ejecutas: python script.py hola 42
sys.argv         # → ['script.py', 'hola', '42']
sys.argv[0]     # → 'script.py'  (nombre del script)
sys.argv[1]     # → 'hola'

# salir del programa con código de estado
sys.exit(0)      # salida exitosa
sys.exit(1)      # salida con error
sys.exit("Error: falta el archivo")  # mensaje + código 1

# versión de Python
sys.version      # → '3.12.0 (main, ...)'
sys.version_info # → sys.version_info(major=3, minor=12, ...)
sys.version_info.major  # → 3

# sistema operativo
sys.platform     # → 'linux' / 'win32' / 'darwin'

# rutas de búsqueda de módulos
sys.path         # lista de directorios donde Python busca módulos
sys.path.append("/mi/modulo")  # agregar ruta personalizada

# tamaño en bytes de un objeto
sys.getsizeof([1, 2, 3])   # → 88  (bytes)

# stdin / stdout / stderr
sys.stdout.write("hola\n")   # igual que print pero sin salto automático
sys.stderr.write("error\n")  # escribir en la salida de error
