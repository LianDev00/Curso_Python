open
open() y with
abrir archivos · context manager · cierre automático

open() abre un archivo y retorna un objeto de archivo. Siempre usa with para abrirlos — garantiza que el archivo se cierre aunque ocurra un error, sin necesidad de llamar f.close() manualmente.

Sin with — peligroso
f = open("datos.txt")
contenido = f.read()
f.close()   # si hay error arriba,
            # esto nunca se ejecuta
            # → archivo queda abierto
Con with — correcto
with open("datos.txt") as f:
    contenido = f.read()
# archivo cerrado automáticamente
# pase lo que pase dentro del with
Parámetros de open()
open(file, mode="r", encoding=None, errors=None)

# encoding — siempre especificarlo para texto
with open("datos.txt", encoding="utf-8") as f: ...

# si no especificas encoding, usa el del sistema operativo
# → puede dar problemas con ñ, tildes en Windows

# errors — qué hacer con caracteres que no se pueden decodificar
with open("datos.txt", encoding="utf-8", errors="ignore") as f: ...
# "strict" (default) → raise UnicodeDecodeError
# "ignore"           → salta caracteres inválidos
# "replace"          → reemplaza con ?

# dos archivos en el mismo with
with open("entrada.txt") as fin, open("salida.txt", "w") as fout:
    fout.write(fin.read())
read
Leer archivos
read · readline · readlines · iteración
Tres formas de leer
# read() — lee TODO el archivo como un solo string
with open("datos.txt", encoding="utf-8") as f:
    todo = f.read()             # string con todo el contenido
    primeros = f.read(100)      # solo los primeros 100 caracteres

# readline() — lee una línea cada vez
with open("datos.txt", encoding="utf-8") as f:
    linea1 = f.readline()       # "primera línea\n"
    linea2 = f.readline()       # "segunda línea\n"

# readlines() — todas las líneas como lista
with open("datos.txt", encoding="utf-8") as f:
    lineas = f.readlines()      # ['línea1\n', 'línea2\n', ...]

# iteración directa — la más eficiente (no carga todo en memoria)
with open("datos.txt", encoding="utf-8") as f:
    for linea in f:                # una línea por iteración
        print(linea.strip())      # .strip() quita el \n
Procesar líneas — patrones comunes
# leer y limpiar todas las líneas
with open("nombres.txt", encoding="utf-8") as f:
    nombres = [l.strip() for l in f if l.strip()]  # ignora líneas vacías

# contar líneas
with open("datos.txt", encoding="utf-8") as f:
    total = sum(1 for _ in f)

# buscar líneas que contienen una palabra
with open("log.txt", encoding="utf-8") as f:
    errores = [l.strip() for l in f if "ERROR" in l]

# posición del cursor
with open("datos.txt", encoding="utf-8") as f:
    f.read(10)
    print(f.tell())     # → 10 (posición actual del cursor)
    f.seek(0)           # volver al inicio
    f.seek(0, 2)        # ir al final (2 = SEEK_END)
¿Cuándo usar cuál?
read() — archivos pequeños que quieres como string completo
for linea in f — archivos grandes, procesamiento línea por línea sin cargar todo
readlines() — cuando necesitas índice o necesitas la lista completa
write
Escribir archivos
write · writelines · append · flush
write() y writelines()
# write() — escribe un string, NO agrega \n automáticamente
with open("salida.txt", "w", encoding="utf-8") as f:
    f.write("Primera línea\n")   # debes agregar \n tú mismo
    f.write("Segunda línea\n")

# writelines() — escribe una lista de strings (tampoco agrega \n)
lineas = ["Ana\n", "Luis\n", "Mía\n"]
with open("nombres.txt", "w", encoding="utf-8") as f:
    f.writelines(lineas)

# agregar al final sin borrar (modo append)
with open("log.txt", "a", encoding="utf-8") as f:
    f.write("Nueva entrada de log\n")

# print() redirigido a archivo — agrega \n automáticamente
with open("salida.txt", "w", encoding="utf-8") as f:
    print("Hola desde print", file=f)
    print("Segunda línea", file=f)

# crear archivo nuevo — falla si ya existe (modo "x")
try:
    with open("nuevo.txt", "x", encoding="utf-8") as f:
        f.write("contenido inicial")
except FileExistsError:
    print("El archivo ya existe")
Modo "w" borra todo lo que había — si el archivo existía, su contenido desaparece al abrirlo en modo "w". Si quieres agregar sin borrar, usa "a". Si quieres protegerte de sobreescritura accidental, usa "x".
!
Manejo de errores con archivos
FileNotFoundError · PermissionError · verificar existencia
Excepciones comunes y cómo manejarlas
import os

# patrón 1: verificar antes de abrir
if os.path.exists("datos.txt"):
    with open("datos.txt", encoding="utf-8") as f:
        contenido = f.read()
else:
    print("Archivo no encontrado")

# patrón 2: try/except (EAFP — más Pythónico)
try:
    with open("datos.txt", encoding="utf-8") as f:
        contenido = f.read()
except FileNotFoundError:
    print("El archivo no existe")
except PermissionError:
    print("Sin permiso para leer el archivo")
except UnicodeDecodeError:
    print("El archivo tiene caracteres que no se pueden decodificar")
except OSError as e:
    print(f"Error de sistema: {e}")

# función robusta de lectura
def leer_archivo(ruta, por_defecto=None):
    try:
        with open(ruta, encoding="utf-8") as f:
            return f.read()
    except FileNotFoundError:
        return por_defecto
csv
Módulo csv
leer · escribir · DictReader · DictWriter

El módulo csv maneja archivos de valores separados por coma. Mucho mejor que dividir strings manualmente — gestiona comillas, comas dentro de valores, y saltos de línea correctamente.

Leer CSV
import csv

# csv.reader — cada fila es una lista
with open("datos.csv", encoding="utf-8") as f:
    lector = csv.reader(f)
    encabezado = next(lector)    # primera fila = encabezado
    for fila in lector:
        print(fila)               # ['Ana', '25', 'Lima']

# csv.DictReader — cada fila es un dict {columna: valor}
with open("datos.csv", encoding="utf-8") as f:
    lector = csv.DictReader(f)
    for fila in lector:
        print(fila["nombre"], fila["edad"])  # acceso por nombre de columna
Escribir CSV
import csv

personas = [
    {"nombre": "Ana",  "edad": 25, "ciudad": "Lima"},
    {"nombre": "Luis", "edad": 30, "ciudad": "Cusco"},
]

# csv.writer — escribir filas como listas
with open("salida.csv", "w", newline="", encoding="utf-8") as f:
    escritor = csv.writer(f)
    escritor.writerow(["nombre", "edad", "ciudad"])  # encabezado
    for p in personas:
        escritor.writerow([p["nombre"], p["edad"], p["ciudad"]])

# csv.DictWriter — escribir dicts directamente
with open("salida.csv", "w", newline="", encoding="utf-8") as f:
    campos   = ["nombre", "edad", "ciudad"]
    escritor = csv.DictWriter(f, fieldnames=campos)
    escritor.writeheader()          # escribe la fila de encabezado
    escritor.writerows(personas)    # escribe todas las filas
newline="" — siempre pásalo cuando escribas CSV en Windows. Sin él, cada fila tendrá una línea en blanco extra porque el módulo csv ya maneja los saltos de línea internamente.
Path
pathlib — rutas orientadas a objetos
Path · operador / · lectura directa · glob

pathlib.Path es la forma moderna de trabajar con rutas. En vez de strings y os.path, usas objetos con métodos. Más legible, más potente, multiplataforma.

pathlib — lo esencial
from pathlib import Path

# crear rutas con operador / (multiplataforma)
base   = Path("/home/lian")
config = base / "proyectos" / "config.json"
# → PosixPath('/home/lian/proyectos/config.json')

# directorio actual y rutas relativas
Path.cwd()              # directorio actual
Path.home()             # directorio home del usuario
Path("datos.txt")       # ruta relativa

# atributos de la ruta
p = Path("/home/lian/datos/archivo.txt")
p.name       # → 'archivo.txt'
p.stem       # → 'archivo'  (sin extensión)
p.suffix     # → '.txt'
p.parent     # → Path('/home/lian/datos')
p.parents[1] # → Path('/home/lian')

# verificar existencia y tipo
p.exists()   # True / False
p.is_file()  # es archivo?
p.is_dir()   # es directorio?

# crear directorios
Path("nueva/carpeta").mkdir(parents=True, exist_ok=True)

# leer y escribir directo desde Path
contenido = Path("datos.txt").read_text(encoding="utf-8")
Path("salida.txt").write_text("contenido", encoding="utf-8")

# glob — buscar archivos con patrón
for archivo in Path(".").glob("*.txt"):
    print(archivo)            # todos los .txt en el directorio actual

for archivo in Path(".").rglob("*.py"):
    print(archivo)            # todos los .py recursivamente

# renombrar y eliminar
Path("viejo.txt").rename("nuevo.txt")
Path("borrar.txt").unlink()         # eliminar archivo
Path("carpeta").rmdir()             # eliminar carpeta vacía
pathlib vs os.path: ambos hacen lo mismo, pero pathlib es más moderno y legible. Path("a") / "b" / "c.txt" vs os.path.join("a", "b", "c.txt"). Para código nuevo, prefiere pathlib.
